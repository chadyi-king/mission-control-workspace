```python
#!/usr/bin/env python3
"""
Agent Base Class - Foundation for all agents
"""

import json
import logging
import time
import sqlite3
from pathlib import Path
from datetime import datetime
from dataclasses import dataclass, asdict
from typing import Optional, Dict, Any, List
from abc import ABC, abstractmethod
import fcntl
import os


@dataclass
class Task:
    """Task definition"""
    id: str
    type: str
    data: Dict[str, Any]
    created_at: datetime
    priority: str = "normal"  # low, normal, high, urgent
    
    @classmethod
    def from_file(cls, path: Path) -> 'Task':
        """Load task from JSON file"""
        data = json.loads(path.read_text())
        return cls(
            id=data['id'],
            type=data['type'],
            data=data.get('data', {}),
            created_at=datetime.fromisoformat(data['timestamp']),
            priority=data.get('priority', 'normal')
        )


@dataclass
class Result:
    """Result definition"""
    task_id: str
    success: bool
    data: Dict[str, Any]
    error: Optional[str] = None
    completed_at: datetime = None
    
    def __post_init__(self):
        if self.completed_at is None:
            self.completed_at = datetime.now()
    
    def to_dict(self) -> Dict:
        """Convert to dictionary for JSON serialization"""
        return {
            'task_id': self.task_id,
            'success': self.success,
            'data': self.data,
            'error': self.error,
            'completed_at': self.completed_at.isoformat()
        }


class AgentBase(ABC):
    """
    Base class for all agents
    
    Provides:
    - Directory management
    - Task processing
    - Error handling
    - Logging
    - State management
    - File locking for concurrency
    """
    
    def __init__(self, name: str, config: Optional[Dict] = None):
        self.name = name
        self.config = config or {}
        
        # Setup paths
        self.workspace = Path.home() / '.openclaw/workspace'
        self.agent_dir = self.workspace / 'agents' / name
        self.inbox = self.agent_dir / 'inbox'
        self.outbox = self.agent_dir / 'outbox'
        self.processed = self.inbox / 'processed'
        self.logs = self.agent_dir / 'logs'
        self.db_dir = self.workspace / 'db'
        
        # Ensure directories exist
        self._setup_directories()
        
        # Setup logging
        self._setup_logging()
        
        # Load or initialize state
        self.state = self._load_state()
        
        # Database connection (optional)
        self.db = None
        if self.config.get('use_database', False):
            self._setup_database()
        
        self.logger.info(f"{name} agent initialized v{self.state.get('version', '1.0.0')}")
    
    def _setup_directories(self):
        """Create all necessary directories"""
        for dir_path in [self.inbox, self.outbox, self.processed, 
                        self.logs, self.db_dir]:
            dir_path.mkdir(parents=True, exist_ok=True)
    
    def _setup_logging(self):
        """Configure logging with rotation"""
        log_file = self.logs / f'{datetime.now():%Y-%m-%d}.log'
        
        # Create formatter
        formatter = logging.Formatter(
            '%(asctime)s - %(name)s - %(levelname)s - [%(filename)s:%(lineno)d] - %(message)s'
        )
        
        # File handler
        file_handler = logging.FileHandler(log_file)
        file_handler.setFormatter(formatter)
        
        # Console handler
        console_handler = logging.StreamHandler()
        console_handler.setFormatter(formatter)
        
        # Setup logger
        self.logger = logging.getLogger(self.name)
        self.logger.setLevel(logging.INFO)
        self.logger.addHandler(file_handler)
        self.logger.addHandler(console_handler)
        
        # Prevent duplicate logs
        self.logger.propagate = False
    
    def _setup_database(self):
        """Setup SQLite database"""
        db_path = self.db_dir / f'{self.name}.db'
        self.db = sqlite3.connect(db_path, check_same_thread=False)
        self.db.row_factory = sqlite3.Row
        self._init_database_schema()
    
    def _init_database_schema(self):
        """Initialize database schema - override in subclass"""
        pass
    
    def _load_state(self) -> Dict:
        """Load or initialize agent state"""
        state_file = self.agent_dir / 'state.json'
        if state_file.exists():
            try:
                return json.loads(state_file.read_text())
            except json.JSONDecodeError as e:
                self.logger.error(f"Failed to parse state: {e}")
                return self._default_state()
        return self._default_state()
    
    def _default_state(self) -> Dict:
        """Default state for new agents"""
        return {
            'agent_name': self.name,
            'status': 'idle',
            'current_task': None,
            'last_heartbeat': datetime.now().isoformat(),
            'tasks_processed': 0,
            'tasks_failed': 0,
            'errors_today': 0,
            'version': '1.0.0',
            'started_at': datetime.now().isoformat()
        }
    
    def _save_state(self):
        """Persist state to disk with file locking"""
        state_file = self.agent_dir / 'state.json'
        temp_file = state_file.with_suffix('.tmp')
        
        # Update timestamp
        self.state['last_heartbeat'] = datetime.now().isoformat()
        
        # Write to temp file first (atomic operation)
        temp_file.write_text(json.dumps(self.state, indent=2))
        
        # Atomic rename
        temp_file.rename(state_file)
    
    def _acquire_lock(self, lock_file: Path) -> bool:
        """Acquire file lock for concurrency control"""
        try:
            fd = os.open(str(lock_file), os.O_CREAT | os.O_RDWR)
            fcntl.flock(fd, fcntl.LOCK_EX | fcntl.LOCK_NB)
            return fd
        except (OSError, IOError):
            return None
    
    def _release_lock(self, fd: int):
        """Release file lock"""
        if fd:
            fcntl.flock(fd, fcntl.LOCK_UN)
            os.close(fd)
    
    def run(self):
        """
        Main execution loop
        Override poll_interval in config
        """
        poll_interval = self.config.get('poll_interval', 60)
        max_errors = self.config.get('max_errors_before_restart', 10)
        
        self.logger.info(f"Starting {self.name} main loop (poll: {poll_interval}s)")
        self.state['status'] = 'active'
        
        try:
            while True:
                self._save_state()
                
                # Check if we should stop
                if self.state['errors_today'] >= max_errors:
                    self.logger.error(f"Too many errors ({max_errors}), stopping")
                    self.state['status'] = 'error'
                    break
                
                # Process inbox
                try:
                    self.process_inbox()
                except Exception as e:
                    self.logger.exception("Error processing inbox")
                    self.state['errors_today'] += 1
                
                # Sleep until next poll
                time.sleep(poll_interval)
                
        except KeyboardInterrupt:
            self.logger.info("Shutdown requested by user")
            self.state['status'] = 'stopped'
        except Exception as e:
            self.logger.exception("Fatal error in main loop")
            self.state['status'] = 'error'
        finally:
            self._save_state()
            if self.db:
                self.db.close()
            self.logger.info(f"{self.name} stopped")
    
    def process_inbox(self):
        """
        Process all tasks in inbox
        Sorted by priority, then by timestamp
        """
        # Get all task files
        task_files = list(self.inbox.glob('*.json'))
        
        # Filter out already processed
        task_files = [f for f in task_files if not f.name.startswith('processed_')]
        
        if not task_files:
            return
        
        self.logger.info(f"Processing {len(task_files)} tasks")
        
        # Sort by priority and timestamp
        tasks_with_files = []
        for task_file in task_files:
            try:
                task = Task.from_file(task_file)
                priority_order = {'urgent': 0, 'high': 1, 'normal': 2, 'low': 3}
                sort_key = (priority_order.get(task.priority, 2), task.created_at)
                tasks_with_files.append((sort_key, task, task_file))
            except Exception as e:
                self.logger.error(f"Failed to parse task {task_file}: {e}")
                self._handle_error(task_file, e)
        
        # Sort and process
        tasks_with_files.sort()
        
        for _, task, task_file in tasks_with_files:
            try:
                # Acquire lock
                lock_file = task_file.with_suffix('.lock')
                fd = self._acquire_lock(lock_file)
                
                if fd is None:
                    self.logger.warning(f"Task {task.id} is locked, skipping")
                    continue
                
                try:
                    # Update state
                    self.state['current_task'] = task.id
                    self.state['status'] = 'processing'
                    self._save_state()
                    
                    # Execute
                    self.logger.info(f"Executing task: {task.id} ({task.type})")
                    result = self.execute(task)
                    
                    # Write result
                    self._write_result(result)
                    
                    # Archive task
                    self._archive_task(task_file)
                    
                    # Update stats
                    if result.success:
                        self.state['tasks_processed'] += 1
                    else:
                        self.state['tasks_failed'] += 1
                    
                    self.logger.info(f"Task {task.id} completed: {'success' if result.success else 'failed'}")
                    
                finally:
                    self._release_lock(fd)
                    if lock_file.exists():
                        lock_file.unlink()
                    
                    self.state['current_task'] = None
                    self.state['status'] = 'active'
                    
            except Exception as e:
                self.logger.exception(f"Error processing task {task.id}")
                self.state['errors_today'] += 1
                self._handle_error(task_file, e)
    
    @abstractmethod
    def execute(self, task: Task) -> Result:
        """
        Execute a task - MUST be implemented by subclasses
        
        This is where your agent's business logic goes
        """
        raise NotImplementedError("Subclasses must implement execute()")
    
    def _write_result(self, result: Result):
        """Write result to outbox"""
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        result_file = self.outbox / f"result_{result.task_id}_{timestamp}.json"
        result_file.write_text(json.dumps(result.to_dict(), indent=2, default=str))
    
    def _archive_task(self, task_file: Path):
        """Move processed task to archive"""
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        archive_name = f"processed_{timestamp}_{task_file.name}"
        task_file.rename(self.processed / archive_name)
    
    def _handle_error(self, task_file: Path, error: Exception):
        """Handle processing errors gracefully"""
        error_file = self.outbox / f"error_{task_file.stem}_{datetime.now():%Y%m%d_%H%M%S}.json"
        error_data = {
            'task_file': str(task_file),
            'error_type': type(error).__name__,
            'error_message': str(error),
            'timestamp': datetime.now().isoformat(),
            'agent': self.name
        }
        error_file.write_text(json.dumps(error_data, indent=2))
        
        # Move task to error folder
        error_dir = self.inbox / 'errors'
        error_dir.mkdir(exist_ok=True)
        task_file.rename(error_dir / task_file.name)


# Example implementation

def execute_with_retry(self, task, max_retries=3):
    for attempt in range(max_retries):
        try:
            return self.execute(task)
        except TransientError as e:
            if attempt < max_retries - 1:
                time.sleep(2 ** attempt)  # Exponential backoff
            else:
                raise

class MessageBus:
    """
    File-based message bus for agent communication
    """
    
    def __init__(self, workspace: Path):
        self.workspace = workspace
        self.agents_dir = workspace / 'agents'
    
    def send(self, from_agent: str, to_agent: str, message: Dict):
        """Send message to agent's inbox"""
        inbox = self.agents_dir / to_agent / 'inbox'
        inbox.mkdir(parents=True, exist_ok=True)
        
        # Add metadata
        message['from'] = from_agent
        message['to'] = to_agent
        message['timestamp'] = datetime.now().isoformat()
        message['id'] = f"MSG-{datetime.now():%Y%m%d%H%M%S}-{uuid4().hex[:8]}"
        
        # Write to inbox
        msg_file = inbox / f"{message['id']}.json"
        msg_file.write_text(json.dumps(message, indent=2))
        
        return message['id']
    
    def receive(self, agent: str, mark_processed: bool = True) -> List[Dict]:
        """Receive all messages for agent"""
        inbox = self.agents_dir / agent / 'inbox'
        
        if not inbox.exists():
            return []
        
        messages = []
        for msg_file in inbox.glob('MSG-*.json'):
            try:
                msg = json.loads(msg_file.read_text())
                messages.append(msg)
                
                if mark_processed:
                    # Move to processed
                    processed_dir = inbox / 'processed'
                    processed_dir.mkdir(exist_ok=True)
                    msg_file.rename(processed_dir / msg_file.name)
                    
            except Exception as e:
                logging.error(f"Failed to read message {msg_file}: {e}")
        
        return messages
```

class StateManager:
    """
    Manages system-wide state
    SQLite is the authoritative source
    JSON is for dashboard (derived)
    """
    
    def __init__(self, db_path: Path):
        self.db = sqlite3.connect(db_path)
        self.init_schema()
    
    def init_schema(self):
        """Initialize database schema"""
        self.db.executescript('''
            CREATE TABLE IF NOT EXISTS tasks (
                id TEXT PRIMARY KEY,
                title TEXT,
                status TEXT,
                priority TEXT,
                agent TEXT,
                created_at TIMESTAMP,
                updated_at TIMESTAMP
            );
            
            CREATE TABLE IF NOT EXISTS agents (
                name TEXT PRIMARY KEY,
                status TEXT,
                last_heartbeat TIMESTAMP,
                current_task TEXT
            );
            
            CREATE INDEX IF NOT EXISTS idx_tasks_status ON tasks(status);
            CREATE INDEX IF NOT EXISTS idx_tasks_agent ON tasks(agent);
        ''')
        self.db.commit()
    
    def sync_to_json(self, output_path: Path):
        """Sync database to JSON for dashboard"""
        data = {
            'lastUpdated': datetime.now().isoformat(),
            'tasks': {},
            'agents': {},
            'stats': {}
        }
        
        # Load tasks
        cursor = self.db.execute('SELECT * FROM tasks')
        for row in cursor.fetchall():
            data['tasks'][row['id']] = dict(row)
        
        # Load agents
        cursor = self.db.execute('SELECT * FROM agents')
        for row in cursor.fetchall():
            data['agents'][row['name']] = dict(row)
        
        # Calculate stats
        cursor = self.db.execute('''
            SELECT status, COUNT(*) as count 
            FROM tasks 
            GROUP BY status
        ''')
        data['stats'] = {row['status']: row['count'] for row in cursor.fetchall()}
        
        # Write to JSON
        output_path.write_text(json.dumps(data, indent=2))

## 13. CHAD_YI Implementation

```python
class CHAD_YI(AgentBase):
    """
    The Face - Human interface layer
    
    Responsibilities:
    - Load context (SOUL, MEMORY, USER)
    - Coordinate workforce
    - Enforce approval workflow
    - Communicate with human
    """
    
    def __init__(self):
        super().__init__('chad-yi', {
            'poll_interval': 5,  # Check frequently for human messages
            'use_database': True
        })
        
        # Load context
        self.context = self.load_full_context()
        
        # Approval workflow
        self.pending_approvals = {}
    
    def load_full_context(self) -> Dict:
        """Load all context files"""
        context = {}
        
        # Core identity
        context['soul'] = self._read_file('SOUL.md')
        context['identity'] = self._read_file('IDENTITY.md')
        
        # User context
        context['user'] = self._read_file('USER.md')
        
        # Long-term memory
        context['memory'] = self._read_file('MEMORY.md')
        
        # Recent events
        today = datetime.now().strftime('%Y-%m-%d')
        yesterday = (datetime.now() - timedelta(days=1)).strftime('%Y-%m-%d')
        context['today'] = self._read_file(f'memory/{today}.md')
        context['yesterday'] = self._read_file(f'memory/{yesterday}.md')
        
        return context
    
    def execute(self, task: Task) -> Result:
        """Handle incoming tasks"""
        if task.type == 'HUMAN_MESSAGE':
            return self.handle_human_message(task)
        elif task.type == 'AGENT_RESULT':
            return self.handle_agent_result(task)
        elif task.type == 'APPROVAL_RESPONSE':
            return self.handle_approval_response(task)
        else:
            return Result(
                task_id=task.id,
                success=False,
                error=f"Unknown task type: {task.type}"
            )
    
    def handle_human_message(self, task: Task) -> Result:
        """Process message from human"""
        message = task.data.get('message', '')
        
        # Parse intent
        intent = self.parse_intent(message)
        
        if intent['type'] == 'task_request':
            # Route to appropriate agent
            target_agent = Router.route(intent['task_type'])
            
            # Create task for agent
            agent_task = {
                'id': f"DELEGATED-{uuid4().hex[:8]}",
                'type': intent['task_type'],
                'data': intent['data'],
                'from': 'chad-yi',
                'timestamp': datetime.now().isoformat()
            }
            
            # Send to agent
            self.message_bus.send('chad-yi', target_agent, agent_task)
            
            return Result(
                task_id=task.id,
                success=True,
                data={'delegated_to': target_agent}
            )
        
        elif intent['type'] == 'status_query':
            # Gather system status
            status = self.gather_system_status()
            
            # Format for human
            formatted = self.format_status(status)
            
            # Send to human (via Telegram or other channel)
            self.send_to_human(formatted)
            
            return Result(
                task_id=task.id,
                success=True,
                data={'status_sent': True}
            )
        
        elif intent['type'] == 'approval_response':
            return self.handle_approval_response(task)
    
    def request_approval(self, proposal: Dict) -> str:
        """
        Request approval from human
        Returns proposal ID
        """
        proposal_id = f"PROP-{datetime.now():%Y%m%d%H%M%S}"
        
        # Format for human
        message = self.format_proposal(proposal)
        
        # Send to human
        self.send_to_human(message)
        
        # Track pending
        self.pending_approvals[proposal_id] = {
            'proposal': proposal,
            'status': 'pending',
            'requested_at': datetime.now()
        }
        
        return proposal_id
```
